@startuml
title Generate User Report - GET /users/report/:user_id

actor Requester
boundary UserGetReportHandler <<userHandler.go>>
control AuthorizeService <<serviceAuthorize.go>>
control UserService <<serviceUser.go>>
control GetFilteredService <<serviceGetFiltered.go>>
entity UserFirestore <<model/user/firestore.go>>
entity ClientFirestore <<model/client/firestore.go>>

Requester -> UserGetReportHandler : GET /users/report/:user_id
activate UserGetReportHandler

UserGetReportHandler -> AuthorizeService : Authorize(Cookie, userID_from_path)
activate AuthorizeService
AuthorizeService -->> UserGetReportHandler : Claims / Error
deactivate AuthorizeService

alt Authorization Failed
    UserGetReportHandler -->> Requester : HTTP Error Response
else Authorization Succeeded
    UserGetReportHandler -> UserService : Report(userID_from_path)
    activate UserService

    UserService -> UserService : Get(userID)
    activate UserService
    UserService -> UserFirestore : Get(&userID)
    activate UserFirestore
    UserFirestore -->> UserService : userInfo / Error
    deactivate UserFirestore
    deactivate UserService

    alt Get UserInfo Failed
        UserService -->> UserGetReportHandler : Error
        UserGetReportHandler -->> Requester : HTTP Error Response
    else Get UserInfo Succeeded
        UserService -> GetFilteredService : GetClientsByUserID(userID)
        activate GetFilteredService
        GetFilteredService -> ClientFirestore : GetAll()
        activate ClientFirestore
        ClientFirestore -->> GetFilteredService : allClients / Error
        deactivate ClientFirestore
        note right of GetFilteredService : Filters clients by userID
        GetFilteredService -->> UserService : clients / Error
        deactivate GetFilteredService

        alt Get Clients Failed
            UserService -->> UserGetReportHandler : Error
            UserGetReportHandler -->> Requester : HTTP Error Response
        else Get Clients Succeeded
            note right of UserService : Assembles UserReport
            UserService -->> UserGetReportHandler : userReport
            UserGetReportHandler -->> Requester : HTTP 200 OK (userReport)
        end
    end
    deactivate UserService
end
deactivate UserGetReportHandler
@enduml
