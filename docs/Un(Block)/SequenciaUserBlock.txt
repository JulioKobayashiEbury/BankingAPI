@startuml
title User Block - PUT /users/:user_id/block

actor Requester
boundary UserPutBlockHandler <<userHandler.go>>
control AuthorizeService <<serviceAuthorize.go>>
control UserService <<serviceUser.go>>
entity UserFirestore <<model/user/firestore.go>>

Requester -> UserPutBlockHandler : PUT /users/:user_id/block
activate Requester
activate UserPutBlockHandler
UserPutBlockHandler -> AuthorizeService : Authorize(Cookie)
activate AuthorizeService
AuthorizeService --> UserPutBlockHandler : Claims / Error
deactivate AuthorizeService
alt Authorization Failed
    UserPutBlockHandler --> Requester : HTTP Error Response
else Authorization Succeeded
    UserPutBlockHandler -> UserService : UserBlock(userID)
    activate UserService
    UserService -> UserService : User(userID)
    UserService -> UserFirestore : Get(&userID)
    activate UserFirestore
    UserFirestore --> UserService : UserData / Error
    deactivate UserFirestore
    alt Get User Failed
        UserService --> UserService : Error
        UserService --> UserPutBlockHandler : Error
        UserPutBlockHandler --> Requester : HTTP Error Response
    else Get User Succeeded
        UserService --> UserService : User
        UserService -> UserFirestore : Update(User with Status=false)
        activate UserFirestore
        UserFirestore --> UserService : nil / Error
        deactivate UserFirestore
        alt Update User Failed
            UserService --> UserPutBlockHandler : Error
            UserPutBlockHandler --> Requester : HTTP Error Response
        else Update User Succeeded
            UserService --> UserPutBlockHandler : nil
            UserPutBlockHandler --> Requester : HTTP 200 OK (User Blocked)
        end
        deactivate UserService
    end
    deactivate UserPutBlockHandler
end
deactivate Requester

@enduml
