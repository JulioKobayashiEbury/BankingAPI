@startuml
title Account UnBlock - PUT /accounts/:account_id/unblock

actor Requester
boundary AccountPutUnblockHandler <<accountHandler.go>>
control AuthorizeService <<serviceAuthorize.go>>
control AccountService <<serviceAccount.go>>
entity AccountFirestore <<model/account/firestore.go>>

Requester -> AccountPutUnblockHandler : PUT /accounts/:account_id/unblock
activate Requester
activate AccountPutUnblockHandler
AccountPutUnblockHandler -> AuthorizeService : Authorize(Cookie)
activate AuthorizeService
AuthorizeService --> AccountPutUnblockHandler : Claims / Error
deactivate AuthorizeService
alt Authorization Failed
    AccountPutUnblockHandler --> Requester : HTTP Error Response
else Authorization Succeeded
   
    AccountPutUnblockHandler -> AccountService : AccountUnblock(accountID)
    activate AccountService
    AccountService -> AccountService : Account(accountID)
    AccountService -> AccountFirestore : Get(&accountID)
    activate AccountFirestore
    AccountFirestore --> AccountService : accountData / Error
    deactivate AccountFirestore
    alt Get Account Failed
        AccountService --> AccountService : Error
        AccountService --> AccountPutUnblockHandler : Error
        AccountPutUnblockHandler --> Requester : HTTP Error Response
    else Get Account Succeeded
        AccountService --> AccountService : account
        AccountService -> AccountFirestore : Update(account with Status=true)
        activate AccountFirestore
        AccountFirestore --> AccountService : nil / Error
        deactivate AccountFirestore
        alt Update Account Failed
            AccountService --> AccountPutUnblockHandler : Error
            AccountPutUnblockHandler --> Requester : HTTP Error Response
        else Update Account Succeeded
            AccountService --> AccountPutUnblockHandler : nil
            AccountPutUnblockHandler --> Requester : HTTP 200 OK (Account Blocked)
        end
        deactivate AccountService
    end
    deactivate AccountPutUnblockHandler
end
deactivate Requester

@enduml
